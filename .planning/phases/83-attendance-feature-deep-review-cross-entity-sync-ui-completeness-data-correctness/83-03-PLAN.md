```typescript
const STATUS_MAP: Record<string, string> = {
  'present': 'הגיע/ה',
  'absent': 'לא הגיע/ה',
  'late': 'איחור'
}

const records = members
  .filter(member => member.status !== 'not_marked')
  .map(member => ({
    studentId: member.id,
    status: STATUS_MAP[member.status] || member.status,
    notes: member.notes || ''
  }))

await apiService.rehearsals.updateAttendance(rehearsal.id, records)
```

Key changes:

3. Remove `arrivalTime` from the payload (not part of the backend schema)
4. The `apiService.rehearsals.updateAttendance(id, records)` signature takes rehearsalId as first arg and records array as second arg — same as AttendanceManager.tsx uses
