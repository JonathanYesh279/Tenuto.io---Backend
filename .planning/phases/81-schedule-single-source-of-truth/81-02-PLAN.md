**Step 2: Create `src/features/students/details/hooks/useStudentScheduleData.ts`**

This hook wraps the API call and transforms the response into the format ScheduleTab and SimpleWeeklyGrid expect:

```typescript
import { useQuery } from '@tanstack/react-query'  // or 'react-query' — match existing codebase import
// Check which import pattern the codebase uses for React Query

interface ScheduleLesson {
  id: string
  instrumentName: string
  teacherName: string
  teacherId?: string
  startTime: string
  endTime: string
  dayOfWeek: number
  location: string | null
  roomNumber?: string | null
  lessonType: 'individual' | 'orchestra' | 'theory'
  isLegacy?: boolean
}

interface WeeklyScheduleResponse {
  individualLessons: any[]
  orchestraRehearsals: any[]
  theoryLessons: any[]
  meta: { studentId: string; totalActivities: number; fetchedAt: string }
}

export function useStudentScheduleData(studentId: string) {
  const query = useQuery({
    queryKey: ['student', studentId, 'weekly-schedule'],
    queryFn: async () => {
      // Use the existing studentDetailsApi or apiService pattern
      // Check which API client the codebase uses for student endpoints
      const response = await studentDetailsApi.getStudentSchedule(studentId)
      return response as WeeklyScheduleResponse
    },
    enabled: !!studentId,
    staleTime: 30_000,  // 30s — schedule data doesn't change frequently
  })

  // Transform backend response into ScheduleLesson[] for SimpleWeeklyGrid
  const lessons: ScheduleLesson[] = useMemo(() => {
    if (!query.data) return []
    const result: ScheduleLesson[] = []

    // Individual lessons — already have all data from backend
    for (const lesson of (query.data.individualLessons || [])) {
      result.push({
        id: lesson.id || lesson.lessonId,
        instrumentName: lesson.instrument || 'כלי נגינה',
        teacherName: lesson.teacherName || 'מורה',
        teacherId: lesson.teacherId,
        startTime: lesson.startTime,
        endTime: lesson.endTime,
        dayOfWeek: lesson.dayOfWeek,
        location: lesson.location,
        roomNumber: null,
        lessonType: 'individual',
        isLegacy: lesson.isLegacy,
      })
    }

    // Orchestra rehearsals
    for (const rehearsal of (query.data.orchestraRehearsals || [])) {
      result.push({
        id: `orchestra-${rehearsal.id || rehearsal.orchestraId}`,
        instrumentName: rehearsal.orchestraName || 'תזמורת',
        teacherName: rehearsal.conductorName || 'מנצח',
        startTime: rehearsal.startTime,
        endTime: rehearsal.endTime,
        dayOfWeek: rehearsal.dayOfWeek,
        location: rehearsal.location,
        roomNumber: null,
        lessonType: 'orchestra',
      })
    }

    // Theory lessons
    for (const theory of (query.data.theoryLessons || [])) {
      result.push({
        id: `theory-${theory.id}`,
        instrumentName: theory.category || 'תאוריה',
        teacherName: theory.teacherName || 'מורה',
        teacherId: theory.teacherId,
        startTime: theory.startTime,
        endTime: theory.endTime,
        dayOfWeek: theory.dayOfWeek,
        location: theory.location,
        roomNumber: null,
        lessonType: 'theory',
      })
    }

    return result
  }, [query.data])

  // Derived data for summary sections
  const personalLessons = useMemo(
    () => lessons.filter(l => l.lessonType !== 'orchestra'),
    [lessons]
  )

  const orchestraActivities = useMemo(
    () => (query.data?.orchestraRehearsals || []).map(r => ({
      id: r.orchestraId || r.id,
      name: r.orchestraName || 'תזמורת',
      conductorName: r.conductorName,
      startTime: r.startTime,
      endTime: r.endTime,
      location: r.location,
      dayOfWeek: r.dayOfWeek,
    })),
    [query.data]
  )

  return {
    lessons,
    personalLessons,
    orchestraActivities,
    isLoading: query.isLoading,
    isError: query.isError,
    error: query.error,
  }
}
```
